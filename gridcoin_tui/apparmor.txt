#include <tunables/global>

  #include <abstractions/base>

  # Capabilities
  file,
  capability,
  network,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within /data
  /data/** rw,

  # Start-up script
  /run.sh rix,

  # Programs the add-on runs.
  /usr/bin/ttyd rix,
  /usr/sbin/nginx rix,
  /usr/bin/gridcoinresearch-tui rix,

  # Terminal devices for ttyd sessions.
  /dev/ptmx rw,
  /dev/pts/* rw,
}
